POL_get_results_of_presidential_elections={
	POL_get_president_popularities = yes
	
	clear_array = POL.candidates_by_vote_share_array
	add_to_array = { POL.candidates_by_vote_share_array = POL.poland_pis_presidential_popularity }
	add_to_array = { POL.candidates_by_vote_share_array = POL.poland_po_presidential_popularity }
	add_to_array = { POL.candidates_by_vote_share_array = POL.poland_right_presidential_popularity }
	add_to_array = { POL.candidates_by_vote_share_array = POL.poland_left_presidential_popularity }
	add_to_array = { POL.candidates_by_vote_share_array = POL.poland_psl_presidential_popularity }
	
	#Check if anyone got an outright majority
	if={
		limit = { check_variable = { var = POL.poland_pis_presidential_popularity value = 0.5 compare = greater_than } }
		if={
			limit = { has_country_flag = Kaczynski_PiS_candidate_flag }
			country_event = poland_election.1002
		}
		else_if={
			limit = { has_country_flag = Duda_PiS_candidate_flag }
			country_event = poland_election.1000
		}
	}
	else_if={
		limit = { check_variable = { var = POL.poland_po_presidential_popularity value = 0.5 compare = greater_than } }
		if={
			limit = { has_country_flag = Komowowski_PO_candidate_flag }
			country_event = poland_election.1100
		}
	}
	else_if={
		limit = { check_variable = { var = POL.poland_right_presidential_popularity value = 0.5 compare = greater_than } }
		if={
			limit = { has_country_flag = Kukiz_RIGHT_candidate_flag }
			country_event = poland_election.1200
		}
	}
	else_if={
		limit = { check_variable = { var = POL.poland_left_presidential_popularity value = 0.5 compare = greater_than } }
		if={
			limit = { has_country_flag = Ogorek_LEFT_candidate_flag }
			country_event = poland_election.1300
		}
		else_if={
			limit = { has_country_flag = Miller_LEFT_candidate_flag }
			country_event = poland_election.1302
		}
	}
	else_if={
		limit = { check_variable = { var = POL.poland_psl_presidential_popularity value = 0.5 compare = greater_than } }
		if={
			limit = { has_country_flag = Jarubas_PSL_candidate_flag }
			country_event = poland_election.1400
		}
	}
	else={
		#No candidate won outright
		set_country_flag = POL_second_round_of_presidential_elections_flag
		
		find_highest_in_array = { array = POL.candidates_by_vote_share_array index = highest_vote_share }
		set_variable = { POL.first_candidate = highest_vote_share }
		remove_from_array = { array = POL.candidates_by_vote_share_array index = POL.first_candidate }
		find_highest_in_array = { array = POL.candidates_by_vote_share_array index = second_highest_vote_share }
		set_variable = { POL.second_candidate = second_highest_vote_share }
		
		clear_array = POL.candidates_by_vote_share_array
		add_to_array = { POL.candidates_by_vote_share_array = POL.poland_pis_presidential_popularity }
		add_to_array = { POL.candidates_by_vote_share_array = POL.poland_po_presidential_popularity }
		add_to_array = { POL.candidates_by_vote_share_array = POL.poland_right_presidential_popularity }
		add_to_array = { POL.candidates_by_vote_share_array = POL.poland_left_presidential_popularity }
		add_to_array = { POL.candidates_by_vote_share_array = POL.poland_psl_presidential_popularity }
		
		if={
			limit = { OR = { 	
				AND = { check_variable = { POL.first_candidate = 0 } check_variable = { POL.second_candidate = 1 } } 
				AND = { check_variable = { POL.first_candidate = 1 } check_variable = { POL.second_candidate = 0 } } } }
			set_country_flag = POL_PiS_vs_PO_president_flag
		}
		else_if={

		}
		else_if={

				AND = { check_variable = { POL.first_candidate = 0 } check_variable = { POL.second_candidate = 3 } } 
				AND = { check_variable = { POL.first_candidate = 3 } check_variable = { POL.second_candidate = 0 } } } }
			set_country_flag = POL_PiS_vs_LEFT_president_flag
		}
		else_if={
			limit = { OR = { 	
				AND = { check_variable = { POL.first_candidate = 0 } check_variable = { POL.second_candidate = 4 } } 
				AND = { check_variable = { POL.first_candidate = 4 } check_variable = { POL.second_candidate = 0 } } } }
			set_country_flag = POL_PiS_vs_PSL_president_flag
		}
