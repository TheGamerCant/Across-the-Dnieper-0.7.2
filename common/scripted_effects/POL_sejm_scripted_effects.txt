POL_bill_needs_simple_majority={
	clr_country_flag = POLAND_bill_needs_simple_majority_flag
	clr_country_flag = POLAND_bill_needs_super_majority_flag
	clr_country_flag = POLAND_bill_needs_three_fifths_majority_flag
	
	set_country_flag = POLAND_bill_needs_simple_majority_flag
	
	set_variable = { POL.POLAND_required_votes = POL.POL_sejm_seat_count }
	divide_variable = { POL.POLAND_required_votes = 2 }
	
	set_variable = { POL.POLAND_required_votes_rounded = POL.POLAND_required_votes }
	round_variable = POL.POLAND_required_votes_rounded
	
	#If there are an odd number of seats in the Sejm, it will round upwards, giving us a simple majority
	#Example: If there are 461 seats, POL.POLAND_required_votes_rounded will be 231, giving us a simple majority
	#		  However, if there are 460 seats for example, POL.POLAND_required_votes_rounded will give us 230, meaning we need to add one
	#		  In this code, you need to use greater than, not greater than or equal to 
	#
	#The same principle applies to any other majorities you may need - make a variable that gives the exact number needed for a majority (two thirds, three fifths, etc.)
	#then duplicate that variable, round it, and check if it is larger than or smaller to the exact number needed, then act accordingly
	
	if={
		limit = { check_variable = { POL.POLAND_required_votes_rounded > POL.POLAND_required_votes } }		#Sejm seat count is odd
		set_variable = { POL.POLAND_required_votes = POL.POLAND_required_votes_rounded }
	}
	else={																									#Sejm seat count is even
		set_variable = { POL.POLAND_required_votes = POL.POLAND_required_votes_rounded }
		add_to_variable = { POL.POLAND_required_votes = 1 }
	}
	clear_variable = POL.POLAND_required_votes_rounded
	
	custom_effect_tooltip = POL_bill_needs_simple_majority_tt
}

	add_to_variable = { THIS.sum_of_popularites = THIS.RIGHT_popularity }
	add_to_variable = { THIS.sum_of_popularites = THIS.PSL_popularity }
	add_to_variable = { THIS.sum_of_popularites = THIS.OTHER_popularity }

	multiply_variable = { THIS.PiS_popularity = 100 }
	multiply_variable = { THIS.LEFT_popularity = 100 }
	multiply_variable = { THIS.RIGHT_popularity = 100 }
	multiply_variable = { THIS.PSL_popularity = 100 }
	multiply_variable = { THIS.OTHER_popularity = 100 }
	
	divide_variable = { THIS.PO_popularity = THIS.sum_of_popularites }
	divide_variable = { THIS.PiS_popularity = THIS.sum_of_popularites }
	divide_variable = { THIS.LEFT_popularity = THIS.sum_of_popularites }
	divide_variable = { THIS.RIGHT_popularity = THIS.sum_of_popularites }
	divide_variable = { THIS.PSL_popularity = THIS.sum_of_popularites }
	divide_variable = { THIS.OTHER_popularity = THIS.sum_of_popularites }
	
	clamp_variable = { var = THIS.PO_popularity min = 0 max = 100 }		#Clamp / correct the variables
	clamp_variable = { var = THIS.PiS_popularity min = 0 max = 100 }
	clamp_variable = { var = THIS.LEFT_popularity min = 0 max = 100 }
	clamp_variable = { var = THIS.RIGHT_popularity min = 0 max = 100 }
	clamp_variable = { var = THIS.PSL_popularity min = 0 max = 100 }
	clamp_variable = { var = THIS.OTHER_popularity min = 0 max = 100 }
	
	clear_variable = THIS.sum_of_popularites

	clear_variable = THIS.LEFT_quotient
	clear_variable = THIS.PSL_quotient
	clear_variable = THIS.OTHER_quotient
	clear_variable = THIS.seats_allocated

POL_can_government_be_formed={
	if={			#PO absolute majority
		limit={ 
			set_temp_variable = { POL_seats_nedded_for_majority = POL_sejm_seat_count }
			divide_temp_variable = { POL_seats_nedded_for_majority = 2 }
			check_variable = { var = POL_PO_seats value = POL_seats_nedded_for_majority compare = greater_than }
		}
		country_event = poland_election.39
	}
	else_if={		#PiS absolute majority
		limit={ 
			set_temp_variable = { POL_seats_nedded_for_majority = POL_sejm_seat_count }
			divide_temp_variable = { POL_seats_nedded_for_majority = 2 }
			check_variable = { var = POL_PiS_seats value = POL_seats_nedded_for_majority compare = greater_than }
		}
		country_event = poland_election.40
	}
	else_if={		#RIGHT absolute majority
		limit={ 
			set_temp_variable = { POL_seats_nedded_for_majority = POL_sejm_seat_count }
			divide_temp_variable = { POL_seats_nedded_for_majority = 2 }
			check_variable = { var = POL_RIGHT_seats value = POL_seats_nedded_for_majority compare = greater_than }
		}
		country_event = poland_election.41
	}
	else_if={		#LEFT absolute majority
		limit={ 
			set_temp_variable = { POL_seats_nedded_for_majority = POL_sejm_seat_count }
			divide_temp_variable = { POL_seats_nedded_for_majority = 2 }
			check_variable = { var = POL_LEFT_seats value = POL_seats_nedded_for_majority compare = greater_than }
		}
		country_event = poland_election.42
	}
	else_if={		#PSL absolute majority
		limit={ 
			set_temp_variable = { POL_seats_nedded_for_majority = POL_sejm_seat_count }
			divide_temp_variable = { POL_seats_nedded_for_majority = 2 }
			check_variable = { var = POL_PSL_seats value = POL_seats_nedded_for_majority compare = greater_than }
		}
		country_event = poland_election.43
	}
	else_if={		#PO, LEFT and PSL combined have a majority
		limit={ 	
			set_temp_variable = { POL.center_coalition = POL.POL_PO_seats }
			add_to_temp_variable = { POL.center_coalition = POL.POL_LEFT_seats }
			add_to_temp_variable = { POL.center_coalition = POL.POL_PSL_seats }
			
			set_temp_variable = { POL_seats_nedded_for_majority = POL_sejm_seat_count }
			divide_temp_variable = { POL_seats_nedded_for_majority = 2 }
			check_variable = { var = POL.center_coalition value = POL_seats_nedded_for_majority compare = greater_than }
		}
		
		if={		#PO leads the coalition
			limit = { 
				check_variable = { var = POL.POL_PO_seats value = POL.POL_PSL_seats compare = greater_than_or_equals }
				check_variable = { var = POL.POL_PO_seats value = POL.POL_LEFT_seats compare = greater_than_or_equals }
			}
			country_event = poland_election.44
		}
		else_if={	#PSL leads the coalition
			limit = { 
				check_variable = { var = POL.POL_PSL_seats value = POL.POL_PO_seats compare = greater_than_or_equals }
				check_variable = { var = POL.POL_PSL_seats value = POL.POL_LEFT_seats compare = greater_than_or_equals }
			}
			country_event = poland_election.48
		}
		else={		#LEFT leads the coalition
			country_event = poland_election.47
		}
	}
	else_if={		#PiS and RIGHT combined have a majority
		limit={ 
			set_temp_variable = { POL.right_coalition = POL.POL_PiS_seats }
			add_to_temp_variable = { POL.right_coalition = POL.POL_RIGHT_seats }	
			set_temp_variable = { POL_seats_nedded_for_majority = POL_sejm_seat_count }
			divide_temp_variable = { POL_seats_nedded_for_majority = 2 }
			check_variable = { var = POL.right_coalition value = POL_seats_nedded_for_majority compare = greater_than }
		}
		
		if={		#PiS leads the coalition
			limit = { check_variable = { var = POL.POL_PiS_seats value = POL.POL_RIGHT_seats compare = greater_than_or_equals } }
			country_event = poland_election.45
		}
		else={		#RIGHT leads the coalition
			country_event = poland_election.46
		}
	}
	else={			#Hung parliament
		if={
			limit={
				set_temp_variable = { POL.center_coalition = POL.POL_PO_seats }
				add_to_temp_variable = { POL.center_coalition = POL.POL_LEFT_seats }
				add_to_temp_variable = { POL.center_coalition = POL.POL_PSL_seats }
				
				set_temp_variable = { POL.right_coalition = POL.POL_PiS_seats }
				add_to_temp_variable = { POL.right_coalition = POL.POL_RIGHT_seats }
				
				check_variable = { var = POL.center_coalition value = POL.right_coalition compare = greater_than_or_equals }
			}
			
			if={		#PO tries to form government
				limit = { 
					check_variable = { var = POL.POL_PO_seats value = POL.POL_PSL_seats compare = greater_than_or_equals }
					check_variable = { var = POL.POL_PO_seats value = POL.POL_LEFT_seats compare = greater_than_or_equals }
				}
				country_event = poland_election.49
			}
			else_if={	#PSL tries to form government
				limit = { 
					check_variable = { var = POL.POL_PSL_seats value = POL.POL_PO_seats compare = greater_than_or_equals }
					check_variable = { var = POL.POL_PSL_seats value = POL.POL_LEFT_seats compare = greater_than_or_equals }
				}
				country_event = poland_election.53
			}
			else={		#LEFT tries to form government
				country_event = poland_election.52
			}
		}
		else={
			if={		#PiS tries to form government
				limit = { 
					check_variable = { var = POL.POL_PiS_seats value = POL.POL_RIGHT_seats compare = greater_than_or_equals }
				}
				country_event = poland_election.50
			}
			else={		#Right tries to form government
				country_event = poland_election.51
			}
		}
	}
}
