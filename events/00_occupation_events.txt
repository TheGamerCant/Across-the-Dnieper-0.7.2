add_namespace = occupation

#Fate of Ukraine
country_event = {
	id = occupation.1
	title = occupation.1.t
	desc = occupation.1.desc

	is_triggered_only = yes

	timeout_days = 20

	option = {
		name = occupation.1.a#Puppet
		
		if={
			limit={
				any_country={
					exists = yes
					original_tag = UKR
					OR={
						is_puppet_of = ROOT
						is_in_faction_with = ROOT
					}
				}
			}
			set_variable = { global.country_releasing_puppet = THIS }
			every_country_with_original_tag = {
				original_tag_to_check = UKR
				limit={
					OR={
						is_puppet_of = ROOT
						is_in_faction_with = ROOT
					}
				}
				set_variable = { global.country_being_puppeted = THIS }
				every_state = {
					limit = { 
						is_owned_and_controlled_by = var:global.country_releasing_puppet
						NOT = { is_claimed_by = var:global.country_releasing_puppet }
						NOT = { is_core_of = PREV.PREV }
						OCCUPATION_is_central_ukraine_trigger = yes
					}
					
					add_core_of = var:global.country_being_puppeted
					var:global.country_being_puppeted = { transfer_state = PREV }
				}
				var:global.country_being_puppeted = { remove_cores = yes }
				clear_variable = global.country_being_puppeted
			}
			clear_variable = global.country_releasing_puppet
		}
		else={
			hidden_effect={
				if={
					limit = {
						original_tag = SOV
						has_government = eastern_conservative
					}
					country_event = { id = russia.67 days = 5 }
				}
			}
			
			if = {
				limit = { UKR = { exists = no } }
				every_state={
					limit = { 
						is_owned_and_controlled_by = ROOT
						NOT = { is_claimed_by = ROOT }
						NOT = { is_core_of = ROOT }
						OCCUPATION_is_central_ukraine_trigger = yes
					}
					add_core_of = UKR
					UKR = { transfer_state = PREV }
				}
				puppet = UKR
			}
			else = {
				create_dynamic_country = {
					original_tag = UKR
					every_state={
						limit = { 
							is_owned_and_controlled_by = PREV.PREV
							NOT = { is_claimed_by = PREV.PREV }
							NOT = { is_core_of = PREV.PREV }
							OCCUPATION_is_central_ukraine_trigger = yes
						}
						add_core_of = PREV
						PREV = { transfer_state = PREV }
					}
					PREV = { puppet = PREV }
					remove_cores = yes
				}
			}
		}	
	}
	
	option = {
		name = occupation.1.b#Annex
		ai_chance = {
			base = 0
		}
		
		add_political_power = -150
		add_stability = -0.1
		add_war_support = -0.1
		every_state={
			limit = {
				is_owned_and_controlled_by = ROOT
				NOT = { is_claimed_by = ROOT }
				NOT = { is_core_of = ROOT }
				OCCUPATION_is_central_ukraine_trigger = yes
			}
			add_claim_by = ROOT
		}
	}
}
