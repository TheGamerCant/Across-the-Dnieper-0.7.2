#Create Konfederacjia
country_event = {
	id = atd_poland.1
	title = atd_poland.1.t
	desc = atd_poland.1.desc
	picture = GFX_report_event_POL_konfederacja
	is_triggered_only = yes

	immediate={
		hidden_effect={
			clr_country_flag = POL_rightist_party_is_kukiz
			set_country_flag = POL_rightist_party_is_konfederacja
			
			set_party_name = { 
				ideology = ultranationalist 
				long_name = POL_KON_no_colour 
				name =  poland_konfederacja_long 
			}
			set_party_name = { 
				ideology = populist 
				long_name = POL_KON_no_colour 
				name =  poland_konfederacja_long 
			}
			set_party_name = { 
				ideology = absolute_monarchy_ideology 
				long_name = POL_KON_no_colour 
				name =  poland_konfederacja_long 
			}
			
			if={
				limit = { bronislaw_komorowski_is_leader = yes }
				POL_make_bonislaw_komorowski_president = yes
			}
			else_if={
				limit = { andrzej_duda_is_leader = yes }
				POL_make_andrzej_duda_president = yes
			}
			else_if={
				limit = { jaroslaw_kaczynski_is_leader = yes }
				POL_make_jaroslaw_kaczynski_president = yes
			}
			
			if={
				limit = { has_idea = POL_PiS_kukiz_coalition }
				swap_ideas={
					remove_idea = POL_PiS_kukiz_coalition
					add_idea = POL_PiS_konfederacja_coalition
				}
			}
		}
	}

	option = {
		name = atd_poland.1.a
		ai_chance = {
			base = 1 
			modifier = {
				factor = 0
				has_country_flag = poland_campaign_as_RIGHT_flag
			}
		}
	}
	
	option = {
		name = atd_poland.1.b
		ai_chance = {
			base = 1 
			modifier = {
				factor = 0
				NOT = { has_country_flag = poland_campaign_as_RIGHT_flag }
			}
		}
		add_popularity = { ideology = ultranationalist popularity = 0.02 }
		
		hidden_effect={
			every_state={
				limit = { POL_is_election_state = yes }
				add_to_variable = { THIS.RIGHT_popularity = 2.5 }
				POL_make_popularities_equal_100 = yes
			}
		}
		
		custom_effect_tooltip = POL_increase_RIGHT_popularity_tt
	}
}
